int findmaxelement(int arr[],int size)
{
    int static i=0,max=-999;
    if(i<size)
    {
        if(max<arr[i])
        {
            max=arr[i];
        }
        i++;
        findmaxelement(arr,size);
    }
    return max;
}
int gcd(int m,int n)
{
    if(n==0)
        return m;
    if(m<n)
        return gcd(n,m);    
    return gcd(n,m%n);
}
void tower(int n,char scr,char aux,char dest)
{
    if(n==1)
    {
        printf("move disk %d from peg %c to peg %c \n",n,scr,dest);
        return ;
    }
    tower(n-1,scr,dest,aux);
    printf("move disk %d from peg %c to peg %c \n",n,scr,dest);
    tower(n-1,aux,scr,dest);
}
